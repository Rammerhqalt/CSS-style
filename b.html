const proxyUrl = 'http://your-proxy-url.com'; // your proxy URL const targetUrl = 'http://target-url.com'; // the URL you want to load via the proxy // Create a new iframe const iframe = document.createElement('iframe'); iframe.src = `${proxyUrl}?url=${ encodeURIComponent(targetUrl) }`; // Set the src attribute to include the target URL iframe.frameBorder = '0'; // Optional: to display the iframe with the default frame border iframe.width = '100%'; // Optional: sets the iframe width iframe.height = '500'; // Optional: sets the iframe height // Add the iframe to the document document.body.appendChild(iframe);
